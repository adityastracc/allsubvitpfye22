//String Reverse

#include <stdio.h>
#include<string.h>
int main()
{
    char s1[10]; 
    int i,n1;
    printf("\nEnter the String: ");
    scanf("%s",s1);
    n1=strlen(s1); 
    printf("\nReverse os string is:");
    for(i=n1-1;i>=0;i--)
    {
        printf("%c",s1[i]);
    }
    printf("\nReverse String using built in function is:%s",strrev(s1));
    return 0;
}
