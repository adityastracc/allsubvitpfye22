#include <stdio.h>
void main()
{

printf("\nEnter Number of Rows:");

{
    for(j=0;j<c;j++)//j=0  j=1  j=2 j=3
    {

    }

for(i=0;i<r;i++)     //i=0  i=1

    for(j=0;j<c;j++)//j=0  j=1  j=2 j=3
    {
        printf(" %d ",m[i][j]);//m[0][0] m[0][1]  m[0][2]
    }

printf("\n The Tranpose of Matrix is:\n ");
